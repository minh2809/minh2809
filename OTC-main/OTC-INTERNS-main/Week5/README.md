     @media screen and (max-width: 1024px) {
     }

   - IMPORTANT !! Tablet devices (landscape tablets, 600px - 768px):

     @media screen and (max-width: 768px) {
     }

   - IMPORTANT !! Mobile devices (phones, 600px - 320px):

     @media screen and (max-width: 600px) {
